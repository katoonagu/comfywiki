# -------- 3) Авторизация Google API --------
from google.colab import auth

from googleapiclient.discovery import build
from googleapiclient.http import MediaFileUpload
drive_api = build("drive","v3")

# -------- 4) Вспомогательные функции (без создания папок) --------
def get_child_folder_id_by_name(name, parent_id):
    """Возвращает ID подпапки name внутри parent_id. Если нет — None.
       Если нашлось несколько — берём первую и предупреждаем."""
    q = (
        f"name = '{name}' and mimeType = 'application/vnd.google-apps.folder' "
        f"and trashed = false and '{parent_id}' in parents"
    )
    res = drive_api.files().list(q=q, spaces="drive", fields="files(id,name)", pageSize=10).execute()
    items = res.get("files", [])
    if not items:
        return None, 0
    if len(items) > 1:
        print(f"⚠️ Найдено несколько папок '{name}' под родителем {parent_id}. Беру первую.")
    return items[0]["id"], len(items)

def resolve_existing_path(segments):
    """Ищет СУЩЕСТВУЮЩИЙ путь относительно 'root' (My Drive). Ошибка, если чего-то нет."""
    parent = "root"  # корень «Мой диск»
    for seg in segments:
        fid, _ = get_child_folder_id_by_name(seg, parent)
        if not fid:
            raise FileNotFoundError(f"❌ Не найдена папка: {' / '.join(segments[:segments.index(seg)+1])}")
        parent = fid
    return parent  # id конечной папки

def find_files_by_name_in_parent(name, parent_id):
    q = f"name = '{name}' and trashed = false and '{parent_id}' in parents"
    res = drive_api.files().list(q=q, spaces="drive", fields="files(id,name,size)", pageSize=100).execute()
    return res.get("files", [])

def aria2_download(link, filename, local_dir="/content"):
    local_path = os.path.join(local_dir, filename)
    print(f"↓ Скачиваю локально: {local_path}")
    cmd = ["aria2c","--continue=true","--file-allocation=none","-x16","-s16","-k1M","-o",filename,link]
    subprocess.run(cmd, check=True)
    if not os.path.exists(local_path):
        raise FileNotFoundError(f"❌ Файл не найден после загрузки: {local_path}")
    size = os.path.getsize(local_path)
    print(f"✅ Локально загружен: {size/1e9:.2f} GB")
    return local_path, size

def resumable_upload(local_path, dest_folder_id, filename, chunk_mb=16):
    file_metadata = {"name": filename, "parents": [dest_folder_id]}
    media = MediaFileUpload(local_path, mimetype="application/octet-stream",
                            resumable=True, chunksize=1024*1024*chunk_mb)
    request = drive_api.files().create(body=file_metadata, media_body=media, fields="id,name,size,webViewLink")
    response = None
    last_print = 0
    print("↑ Загружаю в Google Drive (resumable):", filename)
    while response is None:
        status, response = request.next_chunk()
        if status:
            now = time.time()
            if now - last_print > 1.0:
                print(f"   ... {status.progress()*100:5.1f}%")
                last_print = now
    return response

def list_folder_api(folder_id, max_items=50):
    return drive_api.files().list(
        q=f"'{folder_id}' in parents and trashed=false",
        fields="files(id,name,size,modifiedTime)", pageSize=max_items, orderBy="modifiedTime desc"
    ).execute().get("files", [])

    try:
        if not cfg["LINK"] or cfg["LINK"].startswith("PASTE_URL"):
            raise ValueError(f"Не задан LINK для {cfg['FILENAME']}. Вставь валидный URL.")

        # 1) Проверяем, что нужная подпапка существует (НЕ создаём)
        dest_folder_id = subfolders_ids.get(cfg["SUBFOLDER"])
        if dest_folder_id is None:
            dest_folder_id = resolve_existing_path(TARGET_BASE + [cfg["SUBFOLDER"]])
            subfolders_ids[cfg["SUBFOLDER"]] = dest_folder_id

        # 2) Скачивание
        local_path, size_bytes = aria2_download(cfg["LINK"], cfg["FILENAME"])
        if size_bytes < cfg["MIN_SIZE_GB"] * (1024**3):
            raise RuntimeError(f"❌ Малый размер ({size_bytes/1e9:.2f} GB). MIN={cfg['MIN_SIZE_GB']} GB")

        # 3) Трашим дубликаты (если включено)
        if TRASH_EXISTING:
            for item in find_files_by_name_in_parent(cfg["FILENAME"], dest_folder_id):
                drive_api.files().update(fileId=item["id"], body={"trashed": True}).execute()
                print(f"🗑️  В корзину дубликат: {item['id']} ({int(item.get('size',0))/1e9:.2f} GB)")

        # 4) Загрузка
        resp = resumable_upload(local_path, dest_folder_id, cfg["FILENAME"], CHUNK_MB)
        file_id = resp["id"]
        web = resp.get("webViewLink","(нет ссылки)")
        print(f"✅ Файл создан на Диске: {web}")

        # 5) Проверка размера на Диске
        meta = drive_api.files().get(fileId=file_id, fields="id,name,size,parents").execute()
        gsize = int(meta.get("size","0"))
        print(f"📏 Размер на Диске: {gsize/1e9:.2f} GB")

        results.append({"file": cfg["FILENAME"], "folder": "/".join(TARGET_BASE+[cfg["SUBFOLDER"]]), "id": file_id, "size_gb": gsize/1e9})
    except Exception as e:
        errors.append({"file": cfg["FILENAME"], "err": str(e)})
        print(f"⚠️ Ошибка при обработке {cfg['FILENAME']}: {e}")
